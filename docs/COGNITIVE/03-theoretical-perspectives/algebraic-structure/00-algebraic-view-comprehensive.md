          [op for op in seq if op not in order]
    # Handle V
    if 'V' in seq:
        seq.remove('V')
        seq = ['V'] + seq
    return seq

# Ï† mapping
def phi(seq):
    lat, sec, obs = 0, 0, 0
    for i in range(len(seq)-1):
        l, s, o = get_score(seq[i], seq[i+1])
        lat += l
        sec = min(sec, s) if sec else s
        obs = max(obs, o) if obs else o
    return (lat, sec, obs)

# Example

print('metric:', phi(simplified))
